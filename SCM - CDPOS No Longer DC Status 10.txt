select

right(cdpos.OBJECTID,6),
cdpos.VALUE_NEW,
cdpos.VALUE_OLD,
cdpos.fname,
cdpos.TABKEY,
cdpos.tabname,
CDHDR.username,
cdhdr.tcode,
cdhdr.udate

from CDPOS

inner join cdhdr on
cdhdr.changenr = cdpos.changenr and
cdhdr.objectID = cdpos.objectID

where

cdpos.tabname = 'DMARC'
and cdpos.tabkey like 'GD%'
and cdpos.value_old = '10'
and cdpos.fname = 'MMSTA';
